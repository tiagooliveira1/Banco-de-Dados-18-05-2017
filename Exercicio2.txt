/* Cria tabela Coordenacao */
CREATE TABLE COORDENACAO(
COD_COORDENACAO INTEGER NOT NULL,
NOME VARCHAR2(200),
CONSTRAINT pk_coordenacao PRIMARY KEY(COD_COORDENACAO)
);

/* Cria tabela paciente */
CREATE TABLE PROFESSOR(
COD_PROFESSOR INTEGER NOT NULL,
NOME VARCHAR2(200),
COD_COORDENACAO INTEGER NULL,
CONSTRAINT pk_professor PRIMARY KEY(COD_PROFESSOR),
CONSTRAINT fk_coordenacao_prof FOREIGN KEY (COD_COORDENACAO) REFERENCES COORDENACAO(COD_COORDENACAO)
);

/* Coordenação e médicos */
SELECT C.NOME, P.NOME
FROM COORDENACAO C
INNER JOIN PROFESSOR P ON (P.COD_COORDENACAO = C.COD_COORDENACAO)
ORDER BY C.NOME

/* Coordenação sem professores */
SELECT C.NOME, P.NOME
FROM COORDENACAO C
LEFT JOIN PROFESSOR P ON (P.COD_COORDENACAO = C.COD_COORDENACAO)
WHERE P.COD_COORDENACAO IS NULL
